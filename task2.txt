BITWISE AND , OR and XOR
~~~~~~~~~~~~~~~~~~~~~~~~~~

#include <stdio.h>          
int main()                  
{                           
int n = 8;
int m = 9;
printf("%d", n & m);
return 0;
}

this is the AND operation for which the output was 8.

n=00001000
m=00001001
-------------
  00001000      = 8
  
  
  
  #include <stdio.h>          
int main()                  
{                           
int n = 8;
int m = 9;
printf("%d", n | m);
return 0;
}

this is the OR operation for which the ouput was 9.

n=00001000
m=00001001
----------
  00001001     = 9
  
  
  #include <stdio.h>          
int main()                  
{                           
int n = 8;
int m = 9;
printf("%d", n ^ m);
return 0;
}

this is the XOR operation for which the output was 1.

n=00001000
m=00001001
----------
  00000001   =  1
